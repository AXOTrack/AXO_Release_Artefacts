249 static int cmd_mux_sweep(const struct shell* shell, size_t argc, char** argv) {
250  bool enable = (argc == 2 && strcmp(argv[1], "enable") == 0);
251  if (enable) mux.enableLoad(true);
252  if (setup_adc() < 0) return shell_error(shell, "ADC setup failed"), -EIO;
253 
254  for (uint8_t ch = 1; ch < (1 << mux.numMuxPins()); ++ch) {
255  mux.selectChannel(ch);
256  k_sleep(K_MSEC(50));
257  struct adc_sequence seq = {
258  .channels = BIT(ADC_CHANNEL_ID),
259  .buffer = &adc_buffer,
260  .buffer_size = sizeof(adc_buffer),
261  .resolution = ADC_RESOLUTION,
262  };
263  int ret = adc_read(adc_dev, &seq);
264  if (ret < 0) {
265  shell_error(shell, "ADC read failed on channel %d", ch);
266  continue;
267  }
268  shell_print(shell, "Channel %d: ADC = %d", ch, adc_buffer);
269  val_mV = adc_buffer;
270  ret = adc_raw_to_millivolts(ADC_VREF_MV, ADC_GAIN, ADC_RESOLUTION, &val_mV);
271  __ASSERT_NO_MSG(ret == 0);
272  float val_V = (float) (val_mV / 1000.0f);
273  shell_print(shell, "Channel %d: ADC = %d: mV = %d: V = %f", ch, adc_buffer, val_mV, (double)val_V);
274  k_sleep(K_SECONDS(2));
275  }
276 
277  if (enable) mux.enableLoad(false);
278  shell_print(shell, "Sweep complete");
279  return 0;
280 }
