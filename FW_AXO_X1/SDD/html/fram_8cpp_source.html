531  // compare device id with expected one

533  // if (0) {
534  if ((id[0] != FRAM_DEVICE_ID_1) || (id[1] != FRAM_DEVICE_ID_2) ||
535  (id[2] != FRAM_DEVICE_ID_3) || (id[3] != FRAM_DEVICE_ID_4)) {
536  LOG_ERR("FRAM Selftest: Invalid FRAM device id: %02x %02x %02x %02x",
537  id[0], id[1], id[2], id[3]);
538  } else {
539  // write 2 byte at beginning and end of FRAM, read back and compare
540  uint8_t writePattern[2] = {0x55, 0xAA};
541  uint8_t writePattern2[2] = {0xAA, 0x55};
542  uint8_t readPattern[2];
543  uint32_t addrBeginFram = 0x00;
544  uint32_t addrEndFram = 0x7FFFE;
545  // write 2 byte at beginning and end of FRAM
546  rc = writeBytes(addrBeginFram, &writePattern[0], 2);
547  __ASSERT_NO_MSG(rc == 0);
548  rc = readBytes(addrBeginFram, &readPattern[0], 2);
549  //printk("read: %x, %x\n\n", readPattern[0], readPattern[1]);
550  __ASSERT_NO_MSG(rc == 0);
551  if ((writePattern[0] != readPattern[0]) ||
552  (writePattern[1] != readPattern[1])) {
553  LOG_ERR("FRAM Selftest: Write/Read failed at address: %x",
554  addrBeginFram);
555  return false;
556  }
557  rc = writeBytes(addrEndFram, &writePattern2[0], 2);
558  __ASSERT_NO_MSG(rc == 0);
559  rc = readBytes(addrEndFram, &readPattern[0], 2);
560  __ASSERT_NO_MSG(rc == 0);
561  if ((writePattern2[0] == readPattern[0]) &&
562  (writePattern2[1] == readPattern[1])) {
563  bPassed = true;
564  } else {
565  LOG_ERR("FRAM Selftest: Write/Read failed at address: %x",
566  addrEndFram);
567  }
568  }
569  return bPassed;
