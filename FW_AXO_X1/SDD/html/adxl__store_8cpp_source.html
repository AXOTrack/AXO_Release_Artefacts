1031 void AdxlDataStore::calcMaxXYZ(adxl372_xyz_accel_data* src, uint32_t cnt) {
1032  float x_value, y_value, z_value; // converted axis
1033  int32_t micro_ms2;
1034  uint32_t i;
1035 
1036  for (i = 0; i < cnt; ++i) {
1037  // convert raw values
1038  micro_ms2 = src[i].x * (SENSOR_G / (16 * 1000 / 100));
1039  x_value = static_cast<float>(micro_ms2) / 1000000;
1040  micro_ms2 = src[i].y * (SENSOR_G / (16 * 1000 / 100));
1041  y_value = static_cast<float>(micro_ms2) / 1000000;
1042  micro_ms2 = src[i].z * (SENSOR_G / (16 * 1000 / 100));
1043  z_value = static_cast<float>(micro_ms2) / 1000000;
1044 
1045  // calculate maximum values
1046  if (x_value > x_max) { x_max = x_value; }
1047  if (y_value > y_max) { y_max = y_value; }
1048  if (z_value > z_max) { z_max = z_value; }
1049  }
