315 void RmsTask::task(void) {
316  LOG_INF("RmsTask::task this: %p myName: %s", this, myName);
317 
318  for (;;) {
319  // LOG_INF("%s waiting for messages", myName);
320  PoolElem* pElem = pInMbox->read();
321  if (pElem) {
322  switch (pElem->mtype) {
323  case eAdxl37xSensorData: {
324  // check for 1st data while idle
325  if (idle) {
326  idle = false;
327  init();
328  LOG_DBG("%s", "wakeup");
329  // start idle timer
330  myTimer->start();
331  // notify power manager
332  // it will turn on the boost converter and after 30
333  // seconds request sending measurements over LTE
334  Mbox::sendMsg(msgType::eAdxl37xWake,
335  Mbox::MboxId::eAdxlRms,
336  Mbox::MboxId::ePowerManager);
337  }
338  update_rms(pElem);
339  // indicate that we received data
340  ++rx_cnt;
341  } break;
342 
343  case eTimerExpired:
344  LOG_DBG("%s", "eTimerExpired");
345  pElem = 0; // prevent free()
346  // check for idle condition
347  if (!idle && rx_cnt == rx_cnt_prev) {
348  idle = true;
349  LOG_DBG("%s", "idle");
350  // stop idle timer
351  myTimer->stop();
352  // notify sample task
353  // it will switch back to internal sync
354  Mbox::sendMsg(msgType::eAdxl37xIdle,
355  Mbox::MboxId::eAdxlRms,
356  Mbox::MboxId::eAdxl37xTask);
357  }
358  rx_cnt_prev = rx_cnt;
359  break;
360 
361  case eFwUpdateActive:
362  LOG_WRN("%s", "eFwUpdateActive");
363  break;
364 
365  default:
366  LOG_ERR("unexpected message (%d)", pElem->mtype);
367  }
368  if (pElem) {
369  pElem->free();
370  }
371  }
372  UNIT_TEST_STOP_LOOP;
373  }
