161 
162  for (;;) {

164  PoolElem* pElem = pInMbox->read();
165  if (pElem) {
166  switch (pElem->mtype) {
167  case eAdxl37xRmsData: {
168  LOG_INF("%s", "eAdxl37xRmsData");
169  if (idle) {
170  idle = false;

174 
175  struct adxl_rms_data in =
176  *(struct adxl_rms_data*)pElem->getBuffer();

178 
179  if (!full) {

182  Mbox::send(pElem, Mbox::eHttpConnector);
183  pElem = 0; // prevent free()

185  } break;

187  case eAdxl37xIdle: {

189  idle = true;
190  dump();

192  Mbox::send(pElem, Mbox::eHttpConnector);
