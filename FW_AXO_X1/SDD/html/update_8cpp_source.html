234  LOG_INF("FwUpd::task this: %p myName: %s", this, myName);
235 
236  // This is needed so that MCUBoot won't revert the update
237  auto rc = boot_write_img_confirmed();
238  __ASSERT_NO_MSG(rc == 0);
239 
240  rc = fota_download_init(fota_dl_handler);
241  __ASSERT_NO_MSG(rc == 0);
242 
243  for (;;) {
244  LOG_INF("%s waiting for messages", myName);
245  PoolElem* pElem = pInMbox->read();
246  if (pElem) {
247  switch (pElem->mtype) {
248  case eFwInfo: {
249  // starts download
250  FwInfo info = FwInfo(pElem);
251  rc = info.parse();
252  __ASSERT_NO_MSG(rc == 0);
253  rc = info.validate();
254  __ASSERT_NO_MSG(rc == 0);
255 
256  // get size + chksum and write to download library
257  size_expected = info.getFileSize();
258

265  sha1_expected);
