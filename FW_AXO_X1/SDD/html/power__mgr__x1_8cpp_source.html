229 /******************************************************************************
230  * \brief Power manager X1 message handler function
231  *
232  * \param pElem Pointer to the received message
233  * \return Returns the pointer to the received message
234  *
235  * The message handler function is declared in power manager base class as
236  * pure virtual and therefore it must be must be implemented and overridden
237  * in all derived classes. It is called in the endless loop of the task
238  * function provided by the power manager base class directly after reading
239  * the message from the tasks mailbox. All received messages shall be handled
240  * within this message handler function. \n
241  * In a switch statement all expected messages will be handled, in case of an
242  * unexpected message an error debug message will be triggered. Finally the
243  * pointer pElem will be returned, because it could be modified (set to 0).
244  * \n \n
245  * **Message Handling:** \n
246  * - eMetricsTxTimerExpired: On metrics TX interval timer expiry the pointer
247  * pElem will be set to 0, to prevent releasing the message because it is
248  * allocated static. Resume message will be send to boost converter task by
249  * calling function reqBoostConverter. The reset reason is retrieved by
250  * calling getter function of class ResetReason and initial boost on timer
251  * will be started.
252  * - eBoostCtrlTimerExpired: On initial boost on timer expiry the pointer pElem
253  * will be set to 0, to prevent releasing the message because it is allocated
254  * static. Request messages for an update of temperature and humidity, battery
255  * hardware and LTE metrics will be send by calling function
256  * metricProviderNotifyAll.
257  * - eLteStartDelayTimerExpired: On inital LTE start delay timer expiry the
258  * pointer pElem will be set to 0, to prevent releasing the message because it
259  * is allocated static. A message of type eResume will be send to LTE connector
260  * task by call of function sendMsg provided by Mbox class.
261  * - eAdxlWakeupTimerExpired: On adxl wakeup to measurements transmission timer
262  * expiry the pointer pElem will be set to 0, to prevent releasing the message
263  * because it is allocated static. Measurement send request will be send to
264  * RMS store task by calling function reqRmsReport.
265  * - eLteSleepEnter: If system is not currently booting, indicated by system
266  * uptime greater than timeout value eLteStartTmo, a suspend message will be
267  * send to boost converter task by calling function reqBoostConverter.
268  * - eLteSleepExit: Resume message will be send to boost converter task by
269  * calling function reqBoostConverter.
270  * - eAdxl37xWake: Resume message will be send to boost converter task by
271  * calling function reqBoostConverter and the adxl wakeup to measurements
272  * transmission timer will be started to prepare delayed measurements request.
273  * - eLteConnected: Flag lte_connected will be set to true.
274  * - eLteDisconnected: Flag lte_connected will be set to false.
275  * - eMqttConnected: Reset cause and boot count will be send to data manager
276  * task by calling the functions reportResetCause and reportBootCnt. Afterwards
277  * request messages for an update of temperature and humidity, battery and LTE
278  * and hardware metrics will be send to date manager task by calling function
279  * metricProviderNotifyAll. Finally the metrics TX interval timer will be
280  * started to enable periodic metric reports. \n \n
281  * For more details see also message sequence chart \ref SectionStartup
282  ******************************************************************************/
283 PoolElem* PowerMgrX1::message_handler(PoolElem* pElem) {
284  LOG_INF("PowerMgrX1::message_handler");
285 
286  switch (pElem->mtype) {
287  case eMetricsTxTimerExpired: {
288  LOG_INF("%s", "eMetricsTxTimerExpired");
289  pElem = 0; // prevent free()
290  // prepare tx metrics report by boost on
291  reqBoostConverter(1);
292  // on BoostCtrlTimer timeout (30 sec) lte will be
293  // ready to wake up
294  BoostCtrlTimer->start();
295  } break;
296 
297  case eBoostCtrlTimerExpired: {
298  LOG_INF("%s", "eBoostCtrlTimerExpired");
299  pElem = 0; // prevent free()
300  metricProviderNotifyAll();
301  } break;
302 
303  case eLteStartDelayTimerExpired: {
304  LOG_INF("%s", "eLteStartDelayTimerExpired");
305  pElem = 0; // prevent free()
306  Mbox::sendMsg(msgType::eResume, Mbox::MboxId::ePowerManager,
307  Mbox::MboxId::eLteConnector);
308  } break;
309 
310  case eAdxlWakeupTimerExpired: {
311  LOG_INF("%s", "eAdxlWakeupTimerExpired");
312  pElem = 0; // prevent free()
313  // request measurements to be sent
314  reqRmsReport();
315  } break;
316 
317  case eLteSleepEnter: {
318  LOG_WRN("%s", "eLteSleepEnter");
319  if (k_uptime_get() / 1000 > eLteStartTmo) {
320  // send suspend message to boost converter if not
321  // currently booting
322  reqBoostConverter(0);
323  }
324  } break;
325 
326  case eLteSleepExit: {
327  LOG_WRN("%s", "eLteSleepExit");
328  // send resume message to boost converter
329  reqBoostConverter(1);
330  } break;
331 
332  case eAdxl37xWake: {
333  LOG_WRN("%s", "eAdxl37xWake");
334  // send resume message to boost converter
335  reqBoostConverter(1);
336  // prepare delayed measurements request
337  AdxlWakeupTimer->start();
338  } break;
339 
340  case eLteConnected:
341  LOG_INF("%s", "eLteConnected");
342  lte_connected = true;
343  break;
344 
345  case eLteDisconnected:
346  LOG_INF("%s", "eLteConnected");
347  lte_connected = false;
348  break;
349 
350  case eMqttConnected: {
351  LOG_WRN("%s", "eMqttConnected");
352  // forward reaset reason + boot count
353  reportResetCause();
354  reportBootCnt();
355  // request temperature + humidity, battery
356  // LTE and HW metrics
357  metricProviderNotifyAll();
358  // start periodic reports from now on
359  MetricsTxTimer->start();
360 #ifdef UART_DISABLED
361  LOG_PRINTK("%s", "\n");
362  LOG_WRN("%s", "switching off UART...");
363  k_sleep(K_SECONDS(2));
364  uart0_off();
365 #endif
366  } break;
367 
368  default:
369  LOG_ERR("unexpected message (%d)", pElem->mtype);
370  }
371  return pElem;
372 }
