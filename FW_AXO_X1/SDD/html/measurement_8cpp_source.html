148  } else {

150  bDispCalcFinished = true;
151  LOG_DBG("isComplete %s in %s not found", sVal.c_str(), sDispP2P.c_str());
152  }
153 
154  if ((payload.size() > 30) && bDispCalcFinished) {
155  bRet = true;
156  LOG_DBG("isComplete disp on");

158  }

194  med_hit_cnt.push_back(p->medium_hits);
