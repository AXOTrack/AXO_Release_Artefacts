223 /******************************************************************************
224  * \brief Power manager task function
225  *
226  * Initially the function call virtual task init function from derived
227  * classes, retrieve the reset reason by calling getter function of class
228  * ResetReason. Afterwards the function enters the endless loop and wait
229  * for messages. %Task wakes up when a message is in the mailbox and read
230  * the message by calling mailbox read method which returns the pointer
231  * pElem to the message. If the pointer pElem is not NULL, the virtual
232  * message handler function from derived classes will be called to handle
233  * the received message. Finally the message will be released by
234  * calling class PoolElem method free, if the pointer pElem is not explicitly
235  * set to 0 in message handler function.
236  ******************************************************************************/
237 void PowerMgr::task(void) {
238  LOG_INF("PowerMgr::task this: %p myName: %s", this, myName);
239 
240  taskInit(); // call virtual task init func from derived classes
241 
242  // retrieve reset reason
243  ResetReason::inst()->get(&reason, &reason_str);
244 
245  for (;;) {
246  LOG_DBG("%s waiting for messages", myName);
247  PoolElem* pElem = pInMbox->read();
248  if (pElem) {
249  // call virtual message handler func from derived classes
250  pElem = message_handler(pElem);
251  if (pElem) {
252  pElem->free();
253  }
254  }
255  UNIT_TEST_STOP_LOOP;
256  }
257 }
