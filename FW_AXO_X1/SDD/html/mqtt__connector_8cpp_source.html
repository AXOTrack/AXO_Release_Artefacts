450 void MqttConnector::connect() {
451  LOG_INF("%s", "connect");
452 #ifdef USE_BROKER_SWITCHING
453  enBroker currentBroker = certmgr::inst()->getCurrentBroker();
454 
455  if (currentBroker == eBroker_JOVIA) {
456  strcpy(host_name, JOVIA_MQTT_HOSTNAME);
457  } else if (currentBroker == eBroker_THINGSBOARD) {
458  strcpy(host_name, THINGSBOARD_MQTT_HOSTNAME);
459  } else {
460  LOG_ERR("Unexpected Configuration Current Broker = %d",
461  currentBroker);
462  }
463 #else
464  strcpy(host_name, JOVIA_MQTT_HOSTNAME);
465 #endif //USE_BROKER_SWITCHING
466 
467  static struct mqtt_helper_conn_params conn_params = {0};
468  conn_params.hostname.ptr = host_name;
469  conn_params.hostname.size = strlen(host_name);
470  conn_params.device_id.ptr = device_id;
471  conn_params.device_id.size = strlen(device_id);
472 
473  auto rc = mqtt_helper_connect(&conn_params);
474  LOG_INF("mqtt_helper_connect returned: %d", rc);
475 
476  // prevent inconsistent library states
477  if (rc != 0) {
478  abort();
479  }
480 
481  if (++numConnectRetries < MAX_NUM_CONNECT_RETRIES) {
482  myTimer->start(CONNECT_RETRY_DURATION, CONNECT_RETRY_DURATION);
483  } else {
484 #ifdef USE_BROKER_SWITCHING
485  if (certmgr::inst()->BrokerSwitchRequested()) {
486  LOG_ERR("MQTT connect fails %d times", MAX_NUM_CONNECT_RETRIES);
487  LOG_ERR("Broker switch roll back to JOVIA!");
488  certmgr::inst()->writeBrokerCfg(eBroker_THINGSBOARD, eBroker_JOVIA);
489  ResetCtrl::performSwReset();
490  } else {
491  myTimer->start(metricsInterval, metricsInterval);
492  numConnectRetries = 0;
493  }
494 #else
495  myTimer->start(metricsInterval, metricsInterval);
496  numConnectRetries = 0;
497 #endif //USE_BROKER_SWITCHING
498  }
