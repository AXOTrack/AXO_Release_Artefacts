589 FwFsm::StateId FwDownload::handle_timer_exp() {
590  LOG_INF("%s", "FwDownload::handle_timer_exp");
591 
592  // calculate time since download has started
593  uint32_t now = (uint32_t)(k_uptime_get() / 1000);
594  uint32_t diff = now - startDownloadTimestamp;
595 
596  if ((diff > maxDownloadDelay) ||
597  (reinterpret_cast<FwUpdCtrl*>(pd)->retryCnt >=
598  reinterpret_cast<FwUpdCtrl*>(pd)->maxRetries)) {
599  LOG_ERR("handle_timer_exp: %s", "Download failed, perform Reset!");
600  std::string msg("{\"event\":\"CODE3\"}");
601  Mbox::sendMsg(msgType::eFwDlFailed, Mbox::MboxId::eFwUpdCtrl,
602  Mbox::MboxId::eMqttConnector, msg);
603  k_sleep(K_SECONDS(2));
604  sys_reboot(SYS_REBOOT_COLD);
605  } else {
606  // request dl again
607  reinterpret_cast<FwUpdCtrl*>(pd)->requestDownload();
608  // start a one-shot timer to schedule a dl retry
609  reinterpret_cast<FwUpdCtrl*>(pd)->startRetryTimer();
610  return FwFsm::eNoChange;
611  }
612  return FwFsm::eNoChange; // to avoid warning -Wreturn-type
613 }
