318 int spimram::mram_access(const struct device* spi, struct spi_config* spi_cfg,
319  uint8_t cmd, uint32_t addr, uint8_t* data, size_t len,
320  bool is_write) {
321  LOG_DBG("mram_access cmd: 0x%02x addr: 0x%04x len: %d is_write: %d", cmd,
322  addr, len, is_write);
323 
324  uint8_t header[4] = {cmd, static_cast<uint8_t>((addr >> 16) & 0xFF),
325  static_cast<uint8_t>((addr >> 8) & 0xFF),
326  static_cast<uint8_t>(addr & 0xFF)};
327 
328  struct spi_buf tx_bufs[2] = {
329  {.buf = header, .len = sizeof(header)},
330  {.buf = const_cast<uint8_t*>(data), .len = len}};
331 
332  struct spi_buf rx_bufs[2] = {{.buf = nullptr, .len = sizeof(header)},
333  {.buf = data, .len = len}};
334 
335  struct spi_buf_set tx = {.buffers = tx_bufs, .count = 2};
336  struct spi_buf_set rx = {.buffers = rx_bufs, .count = 2};
337 
338  return is_write ? spi_write(spi, spi_cfg, &tx)
339  : spi_transceive(spi, spi_cfg, &tx, &rx);
340 }

821 void run_mram_stability_test(spimram& mram) {
822  LOG_INF("Starting full MRAM stability test");
823 
824  constexpr uint32_t kTotalSize = 0x400000; // 512k x 8
825  constexpr size_t kChunkSize = 64;
826  uint8_t write_buf[kChunkSize];
827  uint8_t read_buf[kChunkSize];
828 
829  uint32_t iteration = 0;
830  uint32_t total_chunks = kTotalSize / kChunkSize;
831  uint32_t total_cycles = 0;
832  uint32_t write_errors = 0;
833  uint32_t read_errors = 0;
834  uint32_t crc_errors = 0;
835 
836  while (true) {
837  iteration++;
838  LOG_INF("Iteration %u", iteration);
839 
840  for (uint32_t chunk = 0; chunk < total_chunks; ++chunk) {
841  uint32_t addr = chunk * kChunkSize;
842 
843  // Generate test pattern
844  for (size_t i = 0; i < kChunkSize; ++i) {
845  write_buf[i] = (iteration + i + chunk) & 0xFF;
846  }
847 
848  int err = mram.writeBytes(addr, write_buf, kChunkSize);
849  if (err) {
850  LOG_ERR("Write failed at addr 0x%06X (err=%d)", addr, err);
851  write_errors++;
852  continue;
853  }
854 
855  err = mram.readBytes(addr, read_buf, kChunkSize);
856  if (err) {
857  LOG_ERR("Read failed at addr 0x%06X (err=%d)", addr, err);
858  read_errors++;
859  continue;
860  }
861 
862  uint32_t crc_written = crc32_ieee(write_buf, kChunkSize);
863  uint32_t crc_read = crc32_ieee(read_buf, kChunkSize);
864  if (crc_written != crc_read) {
865  LOG_ERR("CRC mismatch at addr 0x%06X", addr);
866  LOG_HEXDUMP_ERR(read_buf, kChunkSize, "Read buffer");
867  crc_errors++;
868  }
869 
870  total_cycles++;
871 
872  if(total_cycles % 5000 == 0) {
873  LOG_INF("Iteration -> %u || Cycles Completed -> %u", iteration, total_cycles);
874  LOG_HEXDUMP_INF(write_buf, kChunkSize, "Write buffer");
875  LOG_HEXDUMP_INF(read_buf, kChunkSize, "Read buffer");
876  }
877  }

879  LOG_INF(

883 
884  total_cycles = 0; // Reset total cycles performed

887  }
