848 void DispTask::calcDisplacement(vectord in, vectord &Vout, vectord &Sout) {
849  uint32_t idx;
850  uint32_t i;
851 
852  idx = rcvdEntries - requestedEntries;
853  if ((idx < NUM_FILT_REMOVE) || (idx > totalNumEntries - NUM_FILT_REMOVE - 1)) {
854  // remove first and last NUM_FILT_REMOVE values (set to 0)
855  Vout[0] = 0.0;
856  Sout[0] = 0.0;
857  } else {
858  Vout[0] = V_last + (DELTA_T * (in[0] + IN_last) / 2);
859  Sout[0] = S_last + (DELTA_T * (Vout[0] + V_last) / 2);
860  detrendCalcSums(dVIdx, Sout[0], detrendVars);
861  dVIdx++;
862  }
863 
864  __ASSERT_NO_MSG(Vout.size() == Sout.size());
865 
866  for (i = 1; i < Vout.size(); i++) {
867  idx = i + (rcvdEntries - requestedEntries);
868  if ((idx < NUM_FILT_REMOVE) || (idx > totalNumEntries - NUM_FILT_REMOVE - 1)) {
869  // remove first and last NUM_FILT_REMOVE values (set to 0)
870  Vout[i] = 0.0;
871  Sout[i] = 0.0;
872  }
873  else {
874  // integrate acceleration to obtain velocity
875  Vout[i] = Vout[i - 1] + (DELTA_T * (in[i] + in[i - 1]) / 2);
876  // integrate velocity to obtain displacement.
877  Sout[i] = Sout[i - 1] + (DELTA_T * (Vout[i] + Vout[i - 1]) / 2);
878  detrendCalcSums(dVIdx, Sout[i], detrendVars);
879  dVIdx++;
880  }
881  }
882  V_last = Vout[i-1];
883  S_last = Sout[i-1];
884  IN_last = in[i-1];
885 }
