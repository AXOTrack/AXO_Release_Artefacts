428  LOG_INF("%s", "psm_read");
429 
430  int ret;
431  ret = nrf_modem_at_scanf("AT+CPSMS?",
432  "+CPSMS: "
433  "%d" /* enabled */
434  "," /* Requested_Periodic-RAU, ignored */
435  "," /* Requested_GPRS-READY-timer, ignored */
436  ",\"%8[0-1]\"" /* Requested_Periodic-TAU */
437  ",\"%8[0-1]\"", /* Requested_Active-Time */
438  &psm_enabled, &request_periodic_tau,
439  &request_active_time);
440 
441  if (ret < 0) {
442  LOG_WRN("Could not parse PSM data, err %d", ret);
443  return;
444  }
445  if (ret > 0) { /* One param matched */
446  LOG_INF("PSM: %s", psm_enabled ? "enabled" : "disabled");
447  }
448  if (ret > 1) { /* Two params matched */
449  LOG_INF("Periodic TAU string: %.*s", sizeof(request_periodic_tau),
450  request_periodic_tau);
451  }
452  if (ret > 2) { /* Three params matched */
453  LOG_INF("Active time string: %.*s", sizeof(request_active_time),
454  request_active_time);
455  }
456 
457  ret = lte_lc_psm_get(&psm_act.tau, &psm_act.active_time);
458  if (ret < 0) {
459  LOG_ERR("lte_lc_psm_get failed, err %d", ret);
460  return;
461  }
462  LOG_INF("granted: tau period: %d active time: %d", psm_act.tau,
463  psm_act.active_time);
