193  LOG_INF("ADConverter::sample chmask: 0x%02x", chmask);
194 
195  int16_t val;
196  int32_t val_mv;
197  uint32_t i;
198  uint8_t bit;
199  auto rc = 0;
200  struct adc_dt_spec *adc_spec;
201 
202  struct adc_sequence sequence = {
203  .buffer = &val,
204  /* buffer size in bytes, not number of samples */
205  .buffer_size = sizeof(val),
206  };
207 
208  for (i = 0U, bit = 1; i < ARRAY_SIZE(adc_channels); i++, bit <<= 1) {
209  #if defined HW_AXO_X1

211  #elif defined HW_AXO_X1_V6

214  }

217  __ASSERT_NO_MSG(rc == 0);
218  adc_spec = &adc_channels[0]; // NRF_SAADC_AIN1
219  }
220  #else
221  #error "Unspecified Hardware"
222  #endif
223  if (bit & chmask) {
224  rc = adc_sequence_init_dt(adc_spec, &sequence);
225  __ASSERT_NO_MSG(rc == 0);
226 
227  k_sleep(K_USEC(500));
228 
229  rc = adc_read(adc_spec->dev, &sequence);
230  __ASSERT_NO_MSG(rc == 0);
231

238  i, val, val_mv);
