642 
643  char buf[40] = {0};
644  snprintf(buf, sizeof(buf), "%d",
645  (int32_t)(k_uptime_get() / 1000)); // convert to seconds
646  payload.append("\"uptime\":" + std::string(buf));
647  payload.append(",\"reset_reason\":\"" + reset_cause + "\"");
648  payload.append(",\"boot_cnt\":" + boot_cnt);
649 
650  payload.append(",\"vbat1\":" + vbat1);
651  payload.append(",\"vbat2\":" + vbat2);
652  payload.append(",\"vdd_det\":" + vdd_det);
653  // payload.append(",\"vdd_int\":" + vdd_int);
654 
655  payload.append(",\"temp\":" + temperature);
656  payload.append(",\"humidity\":" + humidity);
657 
658  payload.append(",\"mdm_batv\":" + mdm_batv);
659  payload.append(",\"mdm_temp\":" + mdm_temp);
660 
661  // if signal strength is not available yet, send "0"
662  if (mdm_link.empty()) mdm_link = "0";
663  payload.append(",\"mdm_link\":" + mdm_link);
664 
665  // if psm is not available yet, send "0"
666  if (psm_tau.empty()) psm_tau = "0";
667  payload.append(",\"psm_tau\":" + psm_tau);
668  if (psm_active.empty()) psm_active = "0";
669  payload.append(",\"psm_active\":" + psm_active);
670 
671  // preliminary include firmware version number
672  payload.append(",\"fw_ver\":");
673  payload.append(std::string(getFwVersion()));
674  payload.append(",\"imsi\":" + imsi);
675 
676  // include configuration version number and configured thresholds
677  payload.append(",\"conf_ver\":\"" + config_version + "\"");
678  payload.append(",\"act_thres\":" + act_thres);
679  payload.append(",\"deact_thres\":" + deact_thres);
680  payload.append(",\"disp_mode_off\":\"" + disp_mode_off + "\"");
681  payload.append(",\"raw_transfer_on\":\"" + raw_transfer_on + "\"");
682  payload.append(",\"disp_transfer_on\":\"" + disp_transfer_on + "\"");
683  payload.append(",\"pub_key\":\"" + pubkey + "\"");
